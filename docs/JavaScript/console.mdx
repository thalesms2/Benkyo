# Comandos para console
```js
console.log("%cTEXT", "color: purple; font-size: 20px")
console.log({testData: 'abc'})
console.error('This is an error')
console.dir(document.body)
const data = {
    id: 1,
    username: "user",
    email: "user@email.com",
    bio: "I am a developer"
}
console.table(data)
console.group() // cria um bloco de log com identação
console.groupEnd()
console.time('function lul')
lul()
console.timeEnd('function lul')
// function lul: 1.283792875393827 ms
console.clear()
```
